```python
# From Uncertainty-Identity Theory mathematical specifications
def uncertainty_identity_vulnerability(individual_data):
    """
    Hogg's uncertainty-identity function: 
    Vulnerability = α(Self_Uncertainty)^β × Group_Entitativity × (1 - Self_Esteem)
    """
    vulnerability = (
        0.73 * (individual_data['self_uncertainty'] ** 1.2) *
        individual_data['group_entitativity'] *
        (1 - individual_data['self_esteem'])
    )
    return min(vulnerability, 1.0)  # Bounded [0,1]

# From Social Learning Theory behavioral equations
def social_learning_influence(network_data, individual_id):
    """
    Bandura's social learning: 
    Influence = Σ(Peer_Behavior × Tie_Strength × Outcome_Observation)
    """
    influence = 0
    for peer_id in network_data.get_neighbors(individual_id):
        peer_behavior = network_data.get_behavior(peer_id)
        tie_strength = network_data.get_tie_strength(individual_id, peer_id)
        outcome_valence = network_data.get_observed_outcome(peer_id)
        influence += peer_behavior * tie_strength * outcome_valence
    return influence / len(network_data.get_neighbors(individual_id))
```

```python
class SpiralOfSilenceAnalyzer:
    def analyze_opinion_expression(self, individual, perceived_majority):
        """
        Noelle-Neumann's spiral process:
        1. Assess perceived opinion climate
        2. Evaluate fear of isolation
        3. Determine expression likelihood
        """

            'climate_assessment': self.assess_perceived_climate(individual),
```
